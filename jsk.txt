  <meta charset="UTF-8">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <title>Number Counting Animation</title>
  <style>
    #counter {
      font-size: 36px;
      font-family: 'Arial', sans-serif;
      text-align: center;
      margin-top: 50px;
      
    }
  </style>
</head>
<body>
  <div id="counter">0</div>

  <script>
    // Function to animate counting from 0 to a target number
    function countUp(target, duration) {
      const start = 0;
      const increment = Math.ceil(target / (duration / 16));
      let current = start;

      const counterElement = document.getElementById('counter');

      function updateCounter() {
        current += increment;
        counterElement.textContent = current;

        if (current < target) {
          requestAnimationFrame(updateCounter);
        } else {
          counterElement.textContent = target;
        }
      }

      updateCounter();
    }

    // Set the target number and duration for the counting animation
    const targetNumber = 100;
    const animationDuration = 2000; // in milliseconds

    // Start the counting animation
    countUp(targetNumber, animationDuration);
  </script>
</body>
